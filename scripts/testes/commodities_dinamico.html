(function() { var gd = document.querySelector('.plotly-graph-div'); gd.on('plotly_relayout', function(ev) { var start = ev['xaxis.range[0]']; if (!start) return; start = new Date(start); gd.data.forEach(function(trace, i) { var xs = trace.x; var orig = trace.meta; var j = xs.findIndex(function(d){ return new Date(d) >= start; }); if (j < 0) return; var base = orig[j]; var yi = orig.map(function(v){ return v/base*100; }); Plotly.restyle(gd, {'y': [yi]}, [i]); }); }); })(); }) };
